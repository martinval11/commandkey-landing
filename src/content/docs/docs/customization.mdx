To customize the CommandKey components, you can use the `className` attribute to change the stylesheet using Tailwind CSS.

```js
<CommandInput className="" />
```

## Example

_App.tsx_
```tsx
import { Command, CommandList, CommandInput, CommandOption } from 'commandkey';
import { useState } from 'react';

import styles from './styles.module.css';

export default function App() {
  const [open, setOpen] = useState(false);

  return (
    <>
      <button onClick={() => setOpen(true)}>Open modal</button>

      <Command
        open={open}
        onClose={() => setOpen(false)}
        className={styles.commandContainer}
      >
        <CommandInput placeholder="Search" className={styles.commandInput} />

        <CommandList className={styles.commandList}>
          <CommandOption className="">Apple</CommandOption>
          <CommandOption className="">Orange</CommandOption>
          <CommandOption className="">Pear</CommandOption>
          <CommandOption className="">Banana</CommandOption>
        </CommandList>
      </Command>
    </>
  );
}
```
