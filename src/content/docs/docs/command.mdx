import CommandLanding from '../../../components/CommandLanding/CommandLanding';

### Import

```js
import { Command } from 'commandkey';
```

### Use

```tsx
<Command open={true} onClose={() => {}}>
  {/* Here your CommandInput, CommandList or CommandOption */}
</Command>
```

### Props

- `open` Open or close the command palette.
- `onClose` Function to be executed when the command closes.
- `className` Edit the input styles using a Tailwind SS class.
- `overlayClassName` Edit the styles for the background behind the Card.
- `children` Inside it uses the input, list and option components.

### Interface

```ts
export interface CommandProps {
  open: boolean;
  onClose: () => void;
  className?: string | any;
  overlayClassName?: string | any;
  children: React.ReactNode;
}
```
